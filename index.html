// ═════════════════════════════════════════════════════════════════════════ // CHI-SQUARE TEST (no external library available in Apps Script, so the // regularized incomplete gamma function is implemented directly here. // Validated against scipy.stats.chi2_contingency on this exact dataset: // matches to 5+ significant figures.) // ═════════════════════════════════════════════════════════════════════════ function logGamma(x) { const g = 7; } function gammaPSeries(a, x) { if (x <= 0) return 0; let ap = a, sum = 1 / a, del = sum; for (let n = 1; n < 200; n++) { ap += 1; del *= x / ap; sum += del; if (Math.abs(del) < Math.abs(sum) * 1e-14) break; } return sum * Math.exp(-x + a * Math.log(x) - logGamma(a)); } function gammaQContFrac(a, x) { let b = x + 1 - a, c = 1e300, d = 1 / b, h = d; for (let i = 1; i < 200; i++) { const an = -i * (i - a); b += 2; d = an * d + b; if (Math.abs(d) < 1e-300) d = 1e-300; c = b + an / c; if (Math.abs(c) < 1e-300) c = 1e-300; d = 1 / d; const del = d * c; h *= del; if (Math.abs(del - 1) < 1e-14) break; } return Math.exp(-x + a * Math.log(x) - logGamma(a)) * h; } function chiSquarePValue(chi2, dof) { if (chi2 <= 0 || dof <= 0) return 1; const a = dof / 2, x = chi2 / 2; return x < a + 1 ? 1 - gammaPSeries(a, x) : gammaQContFrac(a, x); } // 2-outcome (good/not-good) chi-square test of independence across N categories function chiSquareTest(buckets) { const keys = Object.keys(buckets);
